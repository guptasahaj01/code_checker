class GradeBook{
private:
string courseName,instructorName;
public:
GradeBook(){
getline(cin,courseName);
getline(cin,instructorName);
}
void setData(string *s,string *b){
*s=courseName;
*b=instructorName;
}
void displayMessage(){
cout<<"Welcome to the grade book for "<<courseName<<"!\n";
cout<<"This course is presented by: "<<instructorName;
}
}

int main(){
GradeBook stu;
string m,n;
stu.setData(&m,&n);
stu.displayMessage();
}
