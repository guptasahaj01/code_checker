#include <iostream>

#include <cmath>

using namespace std;

class Circle {

float radius;
const float PI=3.14;

public:
Circle(float radius){

    }
    Circle getRadius()
    {
        return radius;
    }
    Circle setRadius(float radius)
    {
        this->radius = radius / PI;
        return radius;
    }
    
    float Area()
    {
        return radius*radius*PI;
    }
    
    float Perimeter()
    {
    
        return 2*radius*PI;
    }
    int equal()
    {
        return Area()==Perimeter();
    
    }

};
int main() {
float r;

    cin >> r;
    Circle c(r);
    cout << c.Perimeter() << endl;
    cout << c.Area() << endl;
    cout << c.equal() <<endl;
    
    
    return 0;

}
